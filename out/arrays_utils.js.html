module.exports = {
 
 /**
  * mess your array
  * @param {array} array - array to mess.
  */
  messeArray: (array) => {    
    let arrayMess = [];
    for(let i = 0; i < array.length - 1; i ++) {          
      let n = Math.floor(Math.random() * array.length);      
      if(arrayMess.length === 0) {
        arrayMess.push(n);  
      }
      if(arrayMess.indexOf(n) === -1) {
        arrayMess.push(n);   
      } else {
         i -= 1;
      }           
    }

    return arrayMess;
  }
};
